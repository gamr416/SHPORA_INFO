n = 5 #сторона квадрата
m = 1 #начальное число
a = [[0 for i in range(n)] for i in range(n)]
for k in range((n + 1) // 2 + 1):
 for j in range(k, n - k):
  a[k][j] = m
  m += 1
 for i in range(k + 1, n - k):
  a[i][n - k - 1] = m
  m += 1
 for j in range(n - k - 2, k - 1, -1):
  a[n - k - 1][j] = m
  m += 1
 for i in range(n - k - 2, k, -1):
  a[i][k] = m
  m += 1
for i in range(n):
 for j in range(n):
  print(a[i][j], end = ' ')
 print()
